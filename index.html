<script>

/*

  Shadow Guesser - single file.

  - 80-word pool (you can edit this array to add your own words).

  - Basic win/lose logic; score per correct reveal; skip and reveal options.

  - Keyboard & onscreen keys supported.

*/

// 80-word pool (single words, lowercase)

const WORDS = [

  "moon", "lilith", "shadow", "raven", "coven", "cauldron", "ember", "sigil",

  "alchemy", "nocturne", "velvet", "sorrow", "cinder", "mirror", "morrow", "phaedra",

  "witchcraft", "crimson", "gossamer", "midnight", "vellichor", "astral", "fable", "thistle",

  "rune", "tincture", "oblivion", "serpent", "orchid", "phantom", "lyre", "onyx",

  "ophelia", "delirium", "tempest", "mystic", "ether", "cinder", "sable", "lament",

  "maw", "emberglow", "gloom", "sable", "eclipse", "covenant", "sorcerer", "banshee",

  "grim", "lamenta", "noir", "saturn", "obsidian", "runeheart", "veil", "sablewood",

  "morrowind", "sepulcher", "hex", "umbra", "gloomwood", "thorn", "wisdom", "silken",

  "dorian", "sablemist", "lure", "ebb", "riptide", "echo", "nymph", "seraph",

  "midwife", "cinderfall", "emberdusk", "harrow", "grisaille", "lilac", "sableborn", "amulet"

];

// Shuffle words once to create a random order for the 80 rounds

let shuffled = WORDS.slice().sort(()=>Math.random()-0.5);

const TOTAL = shuffled.length;

let state = {

  index: 0,

  word: "",

  revealed: [],

  used: new Set(),

  guessesLeft: 8,

  wins: 0,

  losses: 0,

  score: 0

};

const elements = {

  wordContainer: document.getElementById("wordContainer"),

  keyboard: document.getElementById("keyboard"),

  guessesLeft: document.getElementById("guessesLeft"),

  message: document.getElementById("message"),

  lettersUsed: document.getElementById("lettersUsed"),

  wins: document.getElementById("wins"),

  losses: document.getElementById("losses"),

  score: document.getElementById("score"),

  round: document.getElementById("round"),

  currentWordLength: document.getElementById("currentWordLength"),

  progressBar: document.getElementById("progressBar"),

  newBtn: document.getElementById("newBtn"),

  skipBtn: document.getElementById("skipBtn"),

  revealBtn: document.getElementById("revealBtn"),

  prevWord: document.getElementById("prevWord"),

  nextWord: document.getElementById("nextWord")

};

function startRound(i){

  if (i < 0) i = 0;

  if (i >= TOTAL) i = TOTAL-1;

  state.index = i;

  state.word = shuffled[state.index].toLowerCase();

  state.revealed = Array.from(state.word).map(ch => (ch === '-' || ch === ' ') ? ch : "_");

  state.used = new Set();

  state.guessesLeft = 8;

  updateUI();

  renderWord();

  elements.message.textContent = "New round. Choose letters.";

}

function renderWord(){

  elements.wordContainer.innerHTML = "";

  state.revealed.forEach(ch=>{

    const span = document.createElement("div");

    span.className = "letter";

    if (ch === "_"){ span.classList.add("blank"); span.textContent = " "; }

    else span.textContent = ch;

    elements.wordContainer.appendChild(span);

  });

  elements.currentWordLength.textContent = "Length: " + state.word.length;

}

function updateUI(){

  elements.guessesLeft.textContent = state.guessesLeft;

  elements.lettersUsed.textContent = "Letters used: " + (Array.from(state.used).join(", ") || "—");

  elements.wins.textContent = state.wins;

  elements.losses.textContent = state.losses;

  elements.score.textContent = state.score;

  elements.round.textContent = (state.index + 1) + " / " + TOTAL;

  const percent = Math.round(((state.index) / (TOTAL)) * 100);

  elements.progressBar.style.width = percent + "%";

  // update keyboard buttons

  document.querySelectorAll(".key").forEach(k=>{

    const letter = k.dataset.letter;

    if (state.used.has(letter)) k.classList.add("used");

    else k.classList.remove("used");

  });

}

function guessLetter(letter){

  if (state.used.has(letter) || state.guessesLeft <= 0) return;

  state.used.add(letter);

  if (state.word.includes(letter)){

    // reveal positions

    let revealedCount = 0;

    for (let i = 0; i < state.word.length; i++){

      if (state.word[i] === letter){

        state.revealed[i] = letter;

        revealedCount++;

      }

    }

    elements.message.textContent = `Nice — "${letter}" revealed ${revealedCount} time(s).`;

    state.score += 5 * revealedCount;

  } else {

    state.guessesLeft -= 1;

    elements.message.textContent = `Nope — "${letter}" is not in the word.`;

  }

  renderWord();

  updateUI();

  checkEnd();

}

function checkEnd(){

  if (!state.revealed.includes("_")){

    // win

    state.wins += 1;

    state.score += 25;

    elements.message.textContent = "You solved it. Mystic victory.";

    updateUI();

    // auto-advance to next word after a short pause

    setTimeout(()=> startRound(state.index+1), 700);

  } else if (state.guessesLeft <= 0){

    // loss

    state.losses += 1;

    elements.message.textContent = "You lost this round. The word was: " + state.word;

    // reveal word

    state.revealed = Array.from(state.word);

    updateUI();

    renderWord();

    setTimeout(()=> startRound(state.index+1), 900);

  }

}

function createKeyboard(){

  const letters = "abcdefghijklmnopqrstuvwxyz".split("");

  elements.keyboard.innerHTML = "";

  letters.forEach(l=>{

    const btn = document.createElement("button");

    btn.className = "key";

    btn.textContent = l;

    btn.dataset.letter = l;

    btn.addEventListener("click", ()=> guessLetter(l));

    elements.keyboard.appendChild(btn);

  });

}

function revealWord(){

  state.revealed = Array.from(state.word);

  elements.message.textContent = "Revealed. Use this mercy wisely.";

  updateUI();

  renderWord();

}

function skipWord(){

  elements.message.textContent = "Skipped. Moving on.";

  // slight penalty

  state.score = Math.max(0, state.score - 5);

  startRound(state.index+1);

}

// keyboard events

window.addEventListener("keydown", (e)=>{

  if (e.key && e.key.length === 1 && /[a-zA-Z]/.test(e.key)){

    guessLetter(e.key.toLowerCase());

  } else if (e.key === "Enter"){

    // reveal if pressed Enter twice quickly

  } else if (e.key === "ArrowRight"){

    startRound(state.index + 1);

  } else if (e.key === "ArrowLeft"){

    startRound(state.index - 1);

  }

});

// control buttons

elements.newBtn.addEventListener("click", ()=> {

  // shuffle and reset stats

  shuffled = WORDS.slice().sort(()=>Math.random()-0.5);

  state.wins = 0; state.losses = 0; state.score = 0;

  startRound(0);

});

elements.skipBtn.addEventListener("click", skipWord);

elements.revealBtn.addEventListener("click", revealWord);

elements.prevWord.addEventListener("click", ()=> startRound(state.index - 1));

elements.nextWord.addEventListener("click", ()=> startRound(state.index + 1));

// initialize

createKeyboard();

startRound(0);

updateUI();

</script>
